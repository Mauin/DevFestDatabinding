    private static final SparseIntArray sViewsWithIds;

        sViewsWithIds.put(R.id.title_text, 1);
    }

    // views
    public final TextView titleText;

    public ActivityWelcomeBinding(DataBindingComponent bindingComponent, View root) {
        super(bindingComponent, root, 0);
        final Object[] bindings = mapBindings(bindingComponent, root, 4, sIncludes, sViewsWithIds);
        this.titleText = (android.widget.TextView) bindings[1];
        setRootTag(root);
        invalidateAll();
    }
}

@{Expression Language}



  • Math (+, -, *, /)
  • Logical (&&, ||)
  • Comparison (<, >, <=, >=, ==)
Expression Language: Math, Logical, Binary, String concat

Comparison, grouping (), instanceof, null, cast, methods, access arrays/fields

MISSING: this, super, new

@{Expression Language}


  • Literals (strings, numbers)
  • instanceof, null
  • Ternary Operator
  • Method calls
NOT!!! ==> this, super, new
